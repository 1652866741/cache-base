## Usage

Create an instance:


```js

var Cache = require('{%= name %}');

function App() {

}

util.inherits(App, Cache);
```

**Example usage**

```js
var app = new App();

app.set('a', 'b');
app.get('a');
//=> 'b'

app.enable('abc');
console.log(app.enabled('abc'));
//=> 'true'
```
